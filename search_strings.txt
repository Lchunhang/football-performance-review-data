  ("football" OR "soccer" OR "association football")
  AND ("play*" OR "perform*" OR "evaluat*" OR "analys*" OR "metric*" OR "measure*" OR "assessment"
       OR "shoot*" OR "pass*" OR "defend*" OR "dribbl*" OR "goalkeep*" OR "goal*" OR "player")
  AND ("expected goals" OR xg OR "expected assists" OR xa OR "expected threat" OR xt OR vaep
       OR "possession value" OR epv OR "pitch control" OR "space control" OR "pitch value"
       OR "plus-minus" OR PlayeRank OR "pass probability" OR "expected pass*" OR "shot quality")
  AND (event OR tracking OR aggregated OR "player tracking" OR "optical tracking" OR GPS)
  AND NOT ("American football" OR rugby OR futsal OR "Australian Football" OR Gaelic)
  AND NOT (women* OR female* OR girls* OR youth* OR academy OR amateur OR college OR "high school")
)
AND PUBYEAR > 2017 AND PUBYEAR < 2025
AND (LIMIT-TO(LANGUAGE, "English"))
